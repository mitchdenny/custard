// Create a todo app in minutes
var todos = new ListState<string>();

app.Run(() =>
    Border(
        VStack(
            Text("üìù My Todos").Bold().Foreground(Color.Magenta),
            TextBox(input)
                .OnSubmit(text => todos.Add(text))
                .Placeholder("Add a todo..."),
            List(todos)
                .ItemTemplate(todo => 
                    HStack(
                        Text($"‚Ä¢ {todo}"),
                        Button("‚úì", () => todos.Remove(todo))
                    ))
        ).Gap(1).Padding(1)
    ).Title("Todo App").BorderColor(Color.Cyan)
);
